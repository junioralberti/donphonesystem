
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }

    // Helper function to check if user is an admin
    // Assumes a custom claim 'role' is set on the auth token
    function isAdmin() {
      return isAuthenticated() && request.auth.token.role == 'admin';
    }

    // Helper function to check if user is a standard user or admin
    function isUserOrAdmin() {
      return isAuthenticated() && (request.auth.token.role == 'user' || request.auth.token.role == 'admin');
    }

    // Helper function to check if the user is the owner of the document
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // Clients Collection
    match /clients/{clientId} {
      // CHANGED: Now any authenticated user can create, read, update, and delete clients
      allow read, create, update, delete: if isUserOrAdmin();
    }

    // Products Collection
    match /products/{productId} {
      // CHANGED: Now any authenticated user can create, read, update, and delete products
      allow read, create, update, delete: if isUserOrAdmin();
    }

    // Providers Collection
    match /providers/{providerId} {
      // CHANGED: Now any authenticated user can create, read, update, and delete providers
      allow read, create, update, delete: if isUserOrAdmin();
    }

    // Users Collection
    // KEPT AS IS: User management remains restricted for security.
    match /users/{userId} {
      allow read: if isAdmin() || (isAuthenticated() && isOwner(userId));
      allow create: if isAdmin();
      allow update: if isAdmin() || (isAuthenticated() && isOwner(userId));
      allow delete: if isAdmin();
    }

    // Service Orders Collection
    match /serviceOrders/{orderId} {
      // CHANGED: Now any authenticated user can create, read, update, and delete service orders
      allow read, create, update, delete: if isUserOrAdmin();
    }

    // Sales Collection
    match /sales/{saleId} {
      // CHANGED: Now any authenticated user can create, read, update, and delete sales
      allow read, create, update, delete: if isUserOrAdmin();
    }

    // System Settings (Establishment Details)
    match /systemSettings/establishmentDetails {
      // CHANGED: Now any authenticated user with role 'admin' or 'user' can read and write
      allow read, write: if isUserOrAdmin();
    }

    // Default deny all other paths - good practice
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
